**Guest type detection helper (private, inside module):**
```python
def _detect_guest_type(structure) -> str | None:
    """Detect hydrate guest type from atom count or molecule_index."""
    # Try molecule_index first (more reliable)
    if hasattr(structure, 'molecule_index') and structure.molecule_index:
        for entry in structure.molecule_index:
            name = entry.get('mol_type', '').upper()
            if 'CH4' in name:
                return 'ch4'
            elif 'THF' in name:
                return 'thf'
    # Fallback: atom count heuristic
    if hasattr(structure, 'guest_atom_count') and structure.guest_atom_count > 0:
        atoms_per = structure.guest_atom_count / max(structure.guest_nmolecules, 1)
        if abs(atoms_per - 5) < 1:  # CH4 has 5 atoms (united atom)
            return 'ch4'
        elif abs(atoms_per - 13) < 1:  # THF has 13 atoms
            return 'thf'
    return None
```
