```python
from quickice.structure_generation.custom_molecule_inserter import CustomMoleculeInserter
from quickice.structure_generation.types import CustomMoleculeConfig

# Validate files exist
gro_path = Path(self.args.custom_gro)
itp_path = Path(self.args.custom_itp)
if not gro_path.exists():
    print(f"Error: GRO file not found: {gro_path}", file=sys.stderr); return 1
if not itp_path.exists():
    print(f"Error: ITP file not found: {itp_path}", file=sys.stderr); return 1

source = self._interface_result
if source is None:
    print("Error: No interface structure available", file=sys.stderr); return 1

if self.args.custom_placement == "random":
    if self.args.custom_count:
        count = self.args.custom_count
    else:
        liquid_volume = source.water_nmolecules * 0.0299
        count = CustomMoleculeInserter.calculate_molecule_count(self.args.custom_concentration, liquid_volume)
    config = CustomMoleculeConfig(
        gro_path=gro_path, itp_path=itp_path,  # FIX: Pass Path directly, NOT str(gro_path). CustomMoleculeConfig.gro_path is typed as Path.
        placement_mode="random", molecule_count=count
    )
    inserter = CustomMoleculeInserter(config, seed=self.args.seed)
    self._custom_result = inserter.place_random(source, count)
else:  # custom placement
    positions, rotations = self._parse_positions_csv(self.args.custom_positions_file)
    config = CustomMoleculeConfig(
        gro_path=gro_path, itp_path=itp_path,  # FIX: Pass Path directly, NOT str(gro_path).
        placement_mode="custom",
        positions=positions, rotations=rotations  # REQUIRED: __post_init__ validates these are set for custom mode
    )
    inserter = CustomMoleculeInserter(config, seed=self.args.seed)
    self._custom_result = inserter.place_custom(source, positions, rotations)

report_progress(f"Custom molecules: {self._custom_result.custom_molecule_count} placed")
```
