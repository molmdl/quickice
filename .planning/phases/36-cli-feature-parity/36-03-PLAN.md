```python
"""CLI pipeline executor for QuickIce.

Orchestrates multi-step structure generation: Source → Interface → Custom → Solute → Ion → Export.
Mirrors the GUI's MainWindow workflow but runs entirely in the CLI without PySide6/VTK.
"""
import sys
import logging
from pathlib import Path
import argparse

logger = logging.getLogger(__name__)


def report_progress(message: str) -> None:
    """Print progress message to stderr (keeps stdout clean for piping)."""
    print(f"[PROGRESS] {message}", file=sys.stderr)


class CLIPipeline:
    """Orchestrates multi-step structure generation pipeline."""
    
    def __init__(self, args: argparse.Namespace):
        self.args = args
        self._interface_result = None   # InterfaceStructure
        self._hydrate_result = None     # HydrateStructure
        self._custom_result = None      # CustomMoleculeStructure (Plan 04)
        self._solute_result = None      # SoluteStructure (Plan 04)
        self._ion_result = None         # IonStructure (Plan 04)
        self._ice_candidate = None      # Candidate
        self._output_dir = None         # Path — created in execute()
    
    def execute(self) -> int:
        """Run all requested pipeline steps in order. Returns exit code (0=success, 1=error)."""
        # Create output directory
        self._output_dir = Path(self.args.output).resolve()
        self._output_dir.mkdir(parents=True, exist_ok=True)
        
        # Check --no-overwrite
        if getattr(self.args, 'no_overwrite', False):
            existing = list(self._output_dir.glob("*"))
            if existing:
                print(f"Error: Output directory {self._output_dir} is not empty (--no-overwrite set)", file=sys.stderr)
                return 1
        
        # Step 1: Generate source (ice candidate or hydrate)
        if self.args.interface or getattr(self.args, 'hydrate', False):
            exit_code = self._run_source_step()
            if exit_code != 0:
                return exit_code
        
        # Step 2: Interface generation
        if self.args.interface:
            exit_code = self._run_interface_step()
            if exit_code != 0:
                return exit_code
        
        # Step 3: Custom molecule insertion (placeholder — implemented in Plan 04)
        if getattr(self.args, 'custom_gro', None):
            exit_code = self._run_custom_step()
            if exit_code != 0:
                return exit_code
        
        # Step 4: Solute insertion (placeholder — implemented in Plan 04)
        if getattr(self.args, 'solute_type', None):
            exit_code = self._run_solute_step()
            if exit_code != 0:
                return exit_code
        
        # Step 5: Ion insertion (placeholder — implemented in Plan 04)
        if getattr(self.args, 'ion_concentration', None):
            exit_code = self._run_ion_step()
            if exit_code != 0:
                return exit_code
        
        # Step 6: Export
        exit_code = self._run_export_step()
        return exit_code
```

**Step method implementations (3 of 6 in this plan):**

**`_run_source_step() -> int`**: Generate ice candidate or hydrate.
- If `--hydrate`: Create `HydrateConfig` from args, call `HydrateStructureGenerator().generate(config)`, store as `self._hydrate_result`. Report: `"Generated {lattice_type} hydrate with {guest_count} {guest_type} guests"`.
  - HydrateConfig construction:
    ```python
    from quickice.structure_generation.types import HydrateConfig
    config = HydrateConfig(
        lattice_type=getattr(self.args, 'lattice_type', 'sI'),
        guest_type=getattr(self.args, 'guest', 'CH4').lower(),  # CLI uses uppercase CH4/THF but HydrateConfig validates against lowercase GUEST_MOLECULES keys
        supercell_x=getattr(self.args, 'supercell_x', 1),
        supercell_y=getattr(self.args, 'supercell_y', 1),
        supercell_z=getattr(self.args, 'supercell_z', 1),
        cage_occupancy_small=getattr(self.args, 'cage_occupancy_small', 100.0),
        cage_occupancy_large=getattr(self.args, 'cage_occupancy_large', 100.0),
    )
    ```
- If `--interface` without `--hydrate`: Call `generate_candidates()` with `phase_info` from `lookup_phase(args.temperature, args.pressure)`, `nmolecules=256`, `n_candidates=1`. Store first candidate as `self._ice_candidate`. Report: `"Generated ice candidate ({phase_id})"`.
- If `--hydrate` + `--interface`: Generate hydrate first, then convert to candidate via `hydrate.to_candidate()`. Store as `self._ice_candidate`. Report both steps.
- Error handling: try/except with `ImportError` (GenIce2 missing), `ValueError` (bad config). Return 1 on error.
- Use `getattr(self.args, 'hydrate', False)` for safe attribute access.

**`_run_interface_step() -> int`**: Generate interface structure.
- Create `InterfaceConfig` from args (same as existing main.py logic):
  ```python
  from quickice.structure_generation.types import InterfaceConfig
  config = InterfaceConfig(
      mode=self.args.mode,
      box_x=self.args.box_x, box_y=self.args.box_y, box_z=self.args.box_z,
      ice_thickness=self.args.ice_thickness,
      water_thickness=self.args.water_thickness,
      pocket_diameter=getattr(self.args, 'pocket_diameter', None),
      pocket_shape=getattr(self.args, 'pocket_shape', 'sphere'),
      seed=self.args.seed,
  )
  ```
- Source candidate: `self._ice_candidate` (from source step, or hydrate→candidate path)
- Call `generate_interface(self._ice_candidate, config)`.
- Store as `self._interface_result`. Report: `"Interface: {ice_nmolecules} ice + {water_nmolecules} water molecules"`.
- Print report if available: `result.report` to stderr.
- Error handling: try/except with `InterfaceGenerationError`. Return 1 on error.

**`_run_export_step() -> int`**: Export most downstream structure as GROMACS files.
- Determine most downstream structure (priority: ion > solute > custom > interface > ice):
  ```python
  if self._ion_result is not None:
      structure, step_name = self._ion_result, "ion"
  elif self._solute_result is not None:
      structure, step_name = self._solute_result, "solute"
  elif self._custom_result is not None:
      structure, step_name = self._custom_result, "custom"
  elif self._interface_result is not None:
      structure, step_name = self._interface_result, "interface"
  else:
      structure, step_name = self._ice_candidate, "ice"
  ```
- For each structure type, call the appropriate writer functions from `quickice/output/gromacs_writer.py`:
  - Ice candidate: `write_gro_file()`, `write_top_file()`
  - Interface: `write_interface_gro_file()`, `write_interface_top_file()`
  - Custom: `write_custom_molecule_gro_file()`, `write_custom_molecule_top_file()` (Plan 04)
  - Solute: `write_solute_gro_file()`, `write_solute_top_file()` (Plan 04)
  - Ion: `write_ion_gro_file()`, `write_ion_top_file()` (Plan 04)
- Output filenames: `{step_name}.gro`, `{step_name}.top` (per CONTEXT.md decision)
- Copy all ITP files using `quickice/cli/itp_helpers.py:copy_itp_files_for_structure()`
- Report: `"Exported GROMACS files: {step_name}.gro, {step_name}.top, {itp_count} ITP files → {output_dir}"`
- Error handling: try/except with OSError (write failure). Return 1 on error.

```python
@staticmethod
def _parse_positions_csv(filepath: str | Path) -> tuple[list[tuple[float,float,float]], list[tuple[float,float,float]]]:
    """Parse CSV file with custom placement positions.
    
    Expected format: 6 columns (x,y,z,alpha,beta,gamma)
    - x/y/z: center-of-mass position in nm
    - alpha/beta/gamma: Euler angles in degrees (ZXZ convention)
    - Lines starting with # are comments
    - No header row expected
    
    Returns:
        Tuple of (positions, rotations) where each is a list of 3-tuples.
    
    Raises:
        ValueError: If CSV has wrong number of columns or no valid rows
        FileNotFoundError: If file does not exist
    """
    import csv
    filepath = Path(filepath)
    if not filepath.exists():
        raise FileNotFoundError(f"Positions file not found: {filepath}")
    
    positions = []
    rotations = []
    
    with open(filepath) as f:
        reader = csv.reader(line for line in f if not line.strip().startswith('#'))
        for row_num, row in enumerate(reader, 1):
            if not any(row):  # skip blank rows
                continue
            if len(row) != 6:
                raise ValueError(f"Row {row_num} has {len(row)} columns, expected 6 (x,y,z,alpha,beta,gamma)")
            try:
                x, y, z, alpha, beta, gamma = [float(v) for v in row]
            except ValueError as e:
                raise ValueError(f"Row {row_num} has non-numeric value: {e}")
            positions.append((x, y, z))
            rotations.append((alpha, beta, gamma))
    
    if not positions:
        raise ValueError(f"No valid position rows found in {filepath}")
    
    return positions, rotations
```
