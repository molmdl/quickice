The current boundary detection uses polygon.touches(point) which is very precise. For practical use, we should use a small buffer tolerance.

In PhaseDetector.detect_phase(), change the boundary detection logic:

```python
# Instead of exact touches, use distance check with tolerance
BOUNDARY_TOLERANCE = 0.5  # degrees for T, log units for P on log scale

# Check if point is near boundary (within tolerance of multiple phases)
boundary_phases = []
for phase_id, polygon in self._phase_polygons.items():
    # Use distance check instead of touches
    distance = polygon.boundary.distance(point)
    if distance < BOUNDARY_TOLERANCE:
        # Also check if point is close to polygon or inside
        if polygon.contains(point) or polygon.touches(point):
            boundary_phases.append(phase_id)

if len(boundary_phases) >= 2:
    # Format: "Ih/II" or "Multiple phases possible"
    if len(boundary_phases) == 2:
        names = [PHASE_LABELS.get(p, p) for p in boundary_phases]
        return f"{names[0]}/{names[1]}", True
    else:
        return "Multiple phases possible", True
```

This makes boundary clicks more practical by using a tolerance zone.

```python
@Slot()
def _on_input_changed(self):
    """Handle input field changes - update diagram marker."""
    try:
        temp = float(self.input_panel.temp_input.text())
        pressure = float(self.input_panel.pressure_input.text())
        # Only update if values are in valid range
        if 50 <= temp <= 500 and 0.1 <= pressure <= 100000:
            self.diagram_panel.set_coordinates(temp, pressure)
    except (ValueError, TypeError):
        pass  # Invalid input, don't update marker
```
