```python
def _build_custom_guest_info(config) -> dict | None:
    """Build the custom_guest_info dict for the hydrate writers, or None for built-ins.

    Returns {'mol_type': config.guest_type, 'residue_name': '{guest_residue_name}_H',
    'itp_path': Path(config.guest_itp_path)} when config.is_custom_guest, else None.
    """
    if config is None or not getattr(config, "is_custom_guest", False):
        return None
    from pathlib import Path
    return {
        "mol_type": config.guest_type,
        "residue_name": f"{config.guest_residue_name}_H",
        "itp_path": Path(config.guest_itp_path),
    }
```
