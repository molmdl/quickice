```python
"""Command-line argument parser for QuickIce."""
import argparse
from quickice.validation.validators import (
    validate_temperature,
    validate_pressure,
    validate_nmolecules,
)


def get_arguments() -> argparse.Namespace:
    """Parse and validate command-line arguments.
    
    Returns:
        Namespace with validated temperature, pressure, and nmolecules.
    """
    parser = argparse.ArgumentParser(
        description="QuickIce - ML-guided ice structure generation for given thermodynamic conditions",
        prog="quickice",
        formatter_class=argparse.RawDescriptionHelpFormatter,
        epilog="""
Examples:
  python -m quickice.main --temperature 273 --pressure 0.1 --nmolecules 100
  python -m quickice.main --temperature 300 --pressure 100 --nmolecules 256
        """,
    )
    
    parser.add_argument(
        "--temperature",
        type=validate_temperature,
        required=True,
        help="Temperature in Kelvin (valid range: 0-500, inclusive)",
    )
    
    parser.add_argument(
        "--pressure",
        type=validate_pressure,
        required=True,
        help="Pressure in MPa (valid range: 0-10000, inclusive)",
    )
    
    parser.add_argument(
        "--nmolecules",
        type=validate_nmolecules,
        required=True,
        help="Number of water molecules (valid range: 4-100000, inclusive)",
    )
    
    parser.add_argument(
        "--version",
        action="version",
        version="%(prog)s 0.1.0",
    )
    
    return parser.parse_args()
```

```python
"""Main entry point for QuickIce CLI."""
import sys
from quickice.cli.parser import get_arguments


def main() -> int:
    """Main entry point for QuickIce.
    
    Parses arguments and prints validated inputs.
    
    Returns:
        Exit code (0 for success, non-zero for error).
    """
    args = get_arguments()
    
    # For Phase 1, just print the validated inputs
    # Later phases will use these for phase mapping and structure generation
    print(f"Temperature: {args.temperature} K")
    print(f"Pressure: {args.pressure} MPa")
    print(f"Molecules: {args.nmolecules}")
    
    return 0


if __name__ == "__main__":
    sys.exit(main())
```

```python
"""Integration tests for QuickIce CLI."""
import subprocess
import sys


def run_quickice(args: list[str]) -> tuple[int, str, str]:
    """Run quickice CLI with given arguments.
    
    Returns:
        Tuple of (return_code, stdout, stderr).
    """
    result = subprocess.run(
        [sys.executable, "-m", "quickice.main"] + args,
        capture_output=True,
        text=True,
    )
    return result.returncode, result.stdout, result.stderr


class TestCLIIntegration:
    """Integration tests for complete CLI workflow."""
    
    def test_valid_inputs_success(self):
        """Valid inputs should succeed and print values."""
        code, stdout, stderr = run_quickice([
            "--temperature", "300",
            "--pressure", "100",
            "--nmolecules", "256",
        ])
        assert code == 0
        assert "Temperature: 300" in stdout
        assert "Pressure: 100" in stdout
        assert "Molecules: 256" in stdout
    
    def test_valid_boundary_values(self):
        """Boundary values should be accepted."""
        code, stdout, stderr = run_quickice([
            "--temperature", "0",
            "--pressure", "0",
            "--nmolecules", "4",
        ])
        assert code == 0
        assert "Temperature: 0" in stdout
    
    def test_invalid_temperature_rejected(self):
        """Temperature outside range should be rejected."""
        code, stdout, stderr = run_quickice([
            "--temperature", "600",
            "--pressure", "100",
            "--nmolecules", "100",
        ])
        assert code != 0
        assert "temperature" in stderr.lower() or "500" in stderr
    
    def test_invalid_pressure_rejected(self):
        """Pressure outside range should be rejected."""
        code, stdout, stderr = run_quickice([
            "--temperature", "300",
            "--pressure", "-5",
            "--nmolecules", "100",
        ])
        assert code != 0
        assert "pressure" in stderr.lower() or "0" in stderr
    
    def test_invalid_nmolecules_rejected(self):
        """Molecule count outside range should be rejected."""
        code, stdout, stderr = run_quickice([
            "--temperature", "300",
            "--pressure", "100",
            "--nmolecules", "2",
        ])
        assert code != 0
        assert "molecule" in stderr.lower() or "4" in stderr
    
    def test_non_numeric_rejected(self):
        """Non-numeric input should be rejected with clear error."""
        code, stdout, stderr = run_quickice([
            "--temperature", "abc",
            "--pressure", "100",
            "--nmolecules", "100",
        ])
        assert code != 0
        assert "not a valid number" in stderr.lower()
    
    def test_missing_required_flag(self):
        """Missing required flag should show error."""
        code, stdout, stderr = run_quickice([
            "--temperature", "300",
            "--pressure", "100",
            # Missing --nmolecules
        ])
        assert code != 0
        assert "required" in stderr.lower() or "--nmolecules" in stderr
    
    def test_help_flag(self):
        """--help should show usage information."""
        code, stdout, stderr = run_quickice(["--help"])
        assert code == 0
        assert "--temperature" in stdout
        assert "--pressure" in stdout
        assert "--nmolecules" in stdout
    
    def test_version_flag(self):
        """--version should show version."""
        code, stdout, stderr = run_quickice(["--version"])
        assert code == 0
        assert "quickice" in stdout.lower()
```
