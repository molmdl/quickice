    """Command-line argument parser for QuickIce."""
    import argparse
    from quickice.validation.validators import (
        validate_temperature,
        validate_pressure,
        validate_nmolecules,
    )
    
    
    def get_arguments() -> argparse.Namespace:
        """Parse and validate command-line arguments."""
        parser = argparse.ArgumentParser(
            description="QuickIce - ML-guided ice structure generation",
            prog="quickice",
            formatter_class=argparse.RawDescriptionHelpFormatter,
            epilog="""
    Examples:
      quickice --temperature 273 --pressure 0.1 --nmolecules 100
      quickice --temperature 300 --pressure 100 --nmolecules 256
            """,
        )
        
        parser.add_argument(
            "--temperature",
            type=validate_temperature,
            required=True,
            help="Temperature in Kelvin (0-500)",
        )
        
        parser.add_argument(
            "--pressure",
            type=validate_pressure,
            required=True,
            help="Pressure in MPa (0-10000)",
        )
        
        parser.add_argument(
            "--nmolecules",
            type=validate_nmolecules,
            required=True,
            help="Number of water molecules (4-100000)",
        )
        
        parser.add_argument(
            "--version",
            action="version",
            version="%(prog)s 0.1.0",
        )
        
        return parser.parse_args()
    ```

    def main() -> int:
        """Main entry point for QuickIce.
        
        Returns:
            Exit code (0 for success, non-zero for error).
        """
        args = get_arguments()
        
        # Phase 1: Print validated inputs
        # Later phases: phase mapping, GenIce generation, ranking, PDB output

        print(f"Pressure: {args.pressure} MPa")

    """Integration tests for QuickIce CLI."""
    import subprocess
    import sys
    import os
    
    
    def run_quickice(args: list[str]) -> tuple[int, str, str]:
        """Run quickice CLI with given arguments."""
        # Set PYTHONPATH to project root
        env = os.environ.copy()
        env["PYTHONPATH"] = os.getcwd()
        
        result = subprocess.run(
            [sys.executable, "./quickice"] + args,
            capture_output=True,
            text=True,
            env=env,
        )
        return result.returncode, result.stdout, result.stderr
    
    
    class TestCLIIntegration:
        def test_valid_inputs_success(self):
            code, stdout, stderr = run_quickice([
                "--temperature", "300",
                "--pressure", "100",
                "--nmolecules", "256",
            ])
            assert code == 0
            assert "Temperature: 300" in stdout
            assert "Pressure: 100" in stdout
            assert "Molecules: 256" in stdout
        
        def test_valid_boundary_values(self):
            code, stdout, stderr = run_quickice([
                "--temperature", "0",
                "--pressure", "0",
                "--nmolecules", "4",
            ])
            assert code == 0
            assert "Temperature: 0" in stdout
        
        def test_invalid_temperature_rejected(self):
            code, stdout, stderr = run_quickice([
                "--temperature", "600",
                "--pressure", "100",
                "--nmolecules", "100",
            ])
            assert code != 0
            assert "temperature" in stderr.lower() or "500" in stderr
        
        def test_invalid_pressure_rejected(self):
            code, stdout, stderr = run_quickice([
                "--temperature", "300",
                "--pressure", "-5",
                "--nmolecules", "100",
            ])
            assert code != 0
            assert "pressure" in stderr.lower() or "0" in stderr
        
        def test_invalid_nmolecules_rejected(self):
            code, stdout, stderr = run_quickice([
                "--temperature", "300",
                "--pressure", "100",
                "--nmolecules", "2",
            ])
            assert code != 0
            assert "molecule" in stderr.lower() or "4" in stderr
        
        def test_missing_required_flag(self):
            code, stdout, stderr = run_quickice([
                "--temperature", "300",
                "--pressure", "100",
            ])
            assert code != 0
            assert "required" in stderr.lower() or "--nmolecules" in stderr
        
        def test_help_flag(self):
            code, stdout, stderr = run_quickice(["--help"])
            assert code == 0
            assert "--temperature" in stdout
            assert "--pressure" in stdout
            assert "--nmolecules" in stdout
    ```
