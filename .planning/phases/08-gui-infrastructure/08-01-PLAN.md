Create the GenerationWorker class for background ice structure generation using QThread worker pattern.

1. Create `quickice/gui/workers.py` with:

```python
from PySide6.QtCore import QObject, Signal, QThread
from dataclasses import dataclass
from typing import Optional, Any

@dataclass
class GenerationResult:
    """Result from generation worker."""
    success: bool
    result: Optional[Any] = None# The ranked result
    error: Optional[str] = None

class GenerationWorker(QObject):
    """Worker for running ice structure generation in background thread."""
    
    # Signals for progress updates
    progress = Signal(int)      # 0-100 percentage
    status = Signal(str)        # Status message
    finished = Signal(object)   # GenerationResult
    error = Signal(str)         # Error message
    cancelled = Signal()        # Cancellation confirmed
    
    def __init__(self, temperature: float, pressure: float, nmolecules: int):
        super().__init__()
        self._temperature = temperature
        self._pressure = pressure# in bar
        self._nmolecules = nmolecules
        self._is_cancelled = False
    
    def run(self):
        """Execute generation - runs in separate thread."""
        try:
            # Check for cancellation
            if QThread.currentThread().isInterruptionRequested():
                self.cancelled.emit()
                return
            
            # Convert bar to MPa for API calls (1 bar = 0.1 MPa)
            pressure_mpa = self._pressure * 0.1
            
            self.status.emit("Looking up ice phase...")
            self.progress.emit(10)
            
            # Phase lookup
            from quickice.phase_mapping import lookup_phase
            phase_info = lookup_phase(self._temperature, pressure_mpa)
            self.progress.emit(20)
            
            if QThread.currentThread().isInterruptionRequested():
                self.cancelled.emit()
                return
            
            self.status.emit("Generating candidates...")
            
            # Generate candidates
            from quickice.structure_generation import generate_candidates
            gen_result = generate_candidates(
                phase_info=phase_info,
                nmolecules=self._nmolecules,
                n_candidates=10
            )
            self.progress.emit(60)
            
            if QThread.currentThread().isInterruptionRequested():
                self.cancelled.emit()
                return
            
            self.status.emit("Ranking candidates...")
            
            # Ranking
            from quickice.ranking import rank_candidates
            ranking_result = rank_candidates(candidates=gen_result.candidates)
            self.progress.emit(90)
            
            self.status.emit("Complete")
            self.progress.emit(100)
            
            result = GenerationResult(success=True, result=ranking_result)
            self.finished.emit(result)
            
        except Exception as e:
            error_result = GenerationResult(success=False, error=str(e))
            self.error.emit(str(e))
            self.finished.emit(error_result)
```

2. Key implementation notes:
   - Use worker-object pattern (QObject with run method, NOT subclassing QThread)
   - Emit progress at: 10% (start), 20% (phase lookup done), 60% (candidates done), 90% (ranking done), 100% (complete)
   - Status messages: "Looking up ice phase...", "Generating candidates...", "Ranking candidates...", "Complete"
   - Convert pressure from bar (UI) to MPa (API) using `pressure_mpa = pressure * 0.1`
   - Check `QThread.currentThread().isInterruptionRequested()` at each stage for cancellation
   - Import modules INSIDE run() to avoid blocking main thread on import
   - Return GenerationResult dataclass for clean result handling
