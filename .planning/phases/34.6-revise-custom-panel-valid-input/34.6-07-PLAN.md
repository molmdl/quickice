def _update_insert_button_state(self):
    """Update insert button enabled state based on current source availability."""
    if self._current_source == "Interface":
        can_insert = self._interface_structure is not None and self._solute_type is not None
        if not self._interface_structure:
            self.insert_button.setToolTip("Generate Interface first (Tab 2)")
        elif not self._solute_type:
            self.insert_button.setToolTip("Select solute type")
        else:
            self.insert_button.setToolTip(f"Insert {self._solute_type.upper()} solutes")
    
    elif self._current_source == "Custom Molecule":
        can_insert = self._custom_molecule_structure is not None and self._solute_type is not None
        if not self._custom_molecule_structure:
            self.insert_button.setToolTip("Generate Custom Molecules first (Tab 3)")
        elif not self._solute_type:
            self.insert_button.setToolTip("Select solute type")
        else:
            self.insert_button.setToolTip(f"Insert {self._solute_type.upper()} solutes into custom molecule system")
    
    else:
        can_insert = False
        self.insert_button.setToolTip("Select a source structure")
    
    self.insert_button.setEnabled(can_insert)
