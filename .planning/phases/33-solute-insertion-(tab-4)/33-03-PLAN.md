   class SoluteViewerWidget(GenericViewerWidget):
       """3D viewer widget for solute visualization.
       
       Extends GenericViewerWidget with solute-specific rendering.
       """
       
       def __init__(self, parent=None):
           super().__init__(parent)
           logger.info("SoluteViewerWidget initialized")
       
       def render_solute(self, solute_structure):
           """Render solute molecules in 3D viewer.
           
           Args:
               solute_structure: SoluteStructure with positions, atom_names, cell
           """
           from quickice.gui.solute_renderer import create_solute_actor
           
           # Clear existing solute actors
           self.clear_solute_actors()
           
           # Create solute actor
           actor = create_solute_actor(
               solute_structure.positions,
               solute_structure.atom_names,
               solute_structure.cell
           )
           
           if actor:
               self.add_actor(actor, name="solute")
               self.reset_camera()
               logger.info(f"Rendered {solute_structure.n_molecules} solute molecules")
       
       def clear_solute_actors(self):
           """Remove all solute actors from viewer."""
           self.remove_actor_by_name("solute")
   ```

**IMPORTANT:**

    def _update_preview(self):
        """Update molecule count preview in real-time."""
        from quickice.structure_generation.solute_inserter import SoluteInserter

        if self._liquid_volume_nm3 <= 0:
            self.preview_label.setText("No liquid volume")
            return

        concentration = self.concentration_spin.value()

        inserter = SoluteInserter()
        molecule_count = inserter.calculate_molecule_count(concentration, self._liquid_volume_nm3)

        self.preview_label.setText(f"{molecule_count} molecules")
    ```
