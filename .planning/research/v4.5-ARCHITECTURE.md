```python
class SoluteInsertionWorker(QObject):
    """Worker for solute insertion in background thread."""
    
    progress = Signal(int)
    status = Signal(str)
    finished = Signal(object)  # SoluteInsertionResult
    error = Signal(str)
    cancelled = Signal()
    
    def __init__(self, interface_structure, solute_config):
        super().__init__()
        self._interface = interface_structure
        self._config = solute_config
    
    def run(self):
        try:
            if QThread.currentThread().isInterruptionRequested():
                self.cancelled.emit()
                return
            
            self.status.emit("Calculating solute count...")
            self.progress.emit(10)
            
            # Import inside run() to avoid blocking main thread
            from quickice.structure_generation.solute_inserter import SoluteInserter
            
            self.status.emit("Inserting solutes...")
            self.progress.emit(50)
            
            inserter = SoluteInserter(self._config)
            solute_structure = inserter.insert(self._interface)
            
            self.status.emit("Complete")
            self.progress.emit(100)
            
            result = SoluteInsertionResult(success=True, result=solute_structure)
            self.finished.emit(result)
            
        except Exception as e:
            error_result = SoluteInsertionResult(success=False, error=str(e))
            self.error.emit(str(e))
            self.finished.emit(error_result)
```

### 5. SoluteInserter (Domain Layer)

**Location:** `quickice/structure_generation/solute_inserter.py`

**Pattern:** Follow IonInserter

```python
from dataclasses import dataclass
import numpy as np
from scipy.spatial import cKDTree

@dataclass
class SoluteConfig:
    solute_type: str  # "ch4", "thf", "custom"
    concentration_molar: float
    # Additional fields...

@dataclass
class SoluteStructure:
    positions: np.ndarray
    cell: np.ndarray
    molecule_index: list[MoleculeIndex]
    solute_count: int
    solute_type: str
    source_structure: object  # InterfaceStructure

class SoluteInserter:
    """Insert solute molecules into liquid water region."""
    
    # Solute molecule properties
    SOLUTE_INFO = {
        "ch4": {"atoms": 5, "res_name": "CH4", "vdw_radius": 0.19},
        "thf": {"atoms": 12, "res_name": "THF", "vdw_radius": 0.25},
    }
    
    def __init__(self, config: SoluteConfig, seed: int | None = None):
        self.config = config
        self.rng = random.Random(seed)
    
    def calculate_molecule_count(
        self, 
        concentration_molar: float, 
        liquid_volume_nm3: float
    ) -> int:
        """Calculate number of solute molecules from concentration."""
        AVOGADRO = 6.02214076e23
        # concentration (mol/L) × volume (nm³) × (1e-24 L/nm³) × NA
        count = concentration_molar * liquid_volume_nm3 * 1e-24 * AVOGADRO
        return int(count)
    
    def insert(self, interface_structure) -> SoluteStructure:
        """
        Insert solute molecules into liquid water region.
        
        Process:
        1. Identify liquid water molecules from interface structure
        2. Calculate solute count from concentration and volume
        3. Find valid insertion positions (avoid ice, guests, existing solutes)
        4. Replace water molecules with solutes
        5. Build molecule_index with new molecule types
        """
        # Get liquid water molecule indices
        liquid_mols = self._identify_liquid_molecules(interface_structure)
        
        # Calculate solute count
        liquid_volume = self._calculate_liquid_volume(interface_structure)
        solute_count = self.calculate_molecule_count(
            self.config.concentration_molar,
            liquid_volume
        )
        
        # Find valid insertion positions
        valid_positions = self._find_valid_positions(
            interface_structure,
            liquid_mols,
            solute_count,
            min_separation=0.3  # nm
        )
        
        # Replace water with solutes
        new_positions, new_molecule_index = self._replace_water_with_solute(
            interface_structure,
            valid_positions,
            self.config.solute_type
        )
        
        return SoluteStructure(
            positions=new_positions,
            cell=interface_structure.cell,
            molecule_index=new_molecule_index,
            solute_count=solute_count,
            solute_type=self.config.solute_type,
            source_structure=interface_structure
        )
    
    def _identify_liquid_molecules(self, interface_structure) -> list[int]:
        """
        Identify liquid water molecules from InterfaceStructure.
        
        Returns indices of molecules with mol_type == "water"
        """
        liquid_indices = []
        for i, mol in enumerate(interface_structure.molecule_index):
            if mol.mol_type == "water":
                liquid_indices.append(i)
        return liquid_indices
    
    def _find_valid_positions(
        self, 
        structure, 
        liquid_mols, 
        solute_count,
        min_separation
    ) -> list[np.ndarray]:
        """
        Find valid solute insertion positions.
        
        Strategy:
        - Random sampling from liquid region
        - KDTree for distance checking
        - Reject positions too close to ice/guests/existing solutes
        """
        valid_positions = []
        
        # Build KDTree of existing atoms (ice + guests)
        existing_atoms = self._get_existing_atom_positions(structure)
        tree = cKDTree(existing_atoms) if len(existing_atoms) > 0 else None
        
        # Get liquid molecule centers
        liquid_centers = [
            self._get_molecule_center(structure, mol_idx)
            for mol_idx in liquid_mols
        ]
        
        # Random sampling with rejection
        attempts = 0
        max_attempts = solute_count * 100
        
        while len(valid_positions) < solute_count and attempts < max_attempts:
            # Pick random liquid molecule center
            center = self.rng.choice(liquid_centers)
            
            # Add small random displacement
            candidate = center + np.random.uniform(-0.2, 0.2, 3)
            
            # Check distance to existing atoms
            if tree is None or tree.query(candidate, k=1)[0] > min_separation:
                # Check distance to already-placed solutes
                if self._check_distance_to_solute(valid_positions, candidate, min_separation):
                    valid_positions.append(candidate)
            
            attempts += 1
        
        return valid_positions[:solute_count]
```

```python
    """Insert user-provided custom molecule into structure."""
    
    def __init__(self, config: CustomMoleculeConfig):
        self.config = config
        self._parsed_molecule = None
    
    def validate_and_parse(self) -> dict:
        """
        Validate and parse custom molecule files.
        
        Returns:
            dict with:
            - molecule_name: str
            - atom_count: int
            - residue_name: str
            - atom_types: list[str]
            - charges: list[float]
            - valid: bool
            - error: str | None
        """
        try:
            # Parse .gro file
            gro_data = self._parse_gro_file(self.config.gro_path)
            
            # Parse .itp file
            itp_data = self._parse_itp_file(self.config.itp_path)
            
            # Validate consistency
            if gro_data['atom_count'] != itp_data['atom_count']:
                return {
                    'valid': False,
                    'error': f"Atom count mismatch: .gro has {gro_data['atom_count']}, .itp has {itp_data['atom_count']}"
                }
            
            # Store for later use
            self._parsed_molecule = {
                'name': itp_data['molecule_name'],
                'atom_count': itp_data['atom_count'],
                'res_name': itp_data['residue_name'],
                'positions': gro_data['positions'],
                'atom_types': itp_data['atom_types'],
                'charges': itp_data['charges'],
            }
            
            return {'valid': True, **self._parsed_molecule}
            
        except Exception as e:
            return {'valid': False, 'error': str(e)}
    
    def insert(self, interface_structure) -> CustomMoleculeStructure:
        """
        Insert custom molecule into structure.
        
        Placement modes:
        1. "random": Place at random valid position in liquid region
        2. "custom": Place at user-specified position (with PBC handling)
        """
        if self._parsed_molecule is None:
            raise ValueError("Must call validate_and_parse() first")
        
        if self.config.placement_mode == "random":
            position = self._find_random_valid_position(interface_structure)
        else:
            position = self.config.custom_position
        
        # Apply PBC wrapping if needed
        wrapped_position = self._apply_pbc(position, interface_structure.cell)
        
        # Insert molecule at position
        new_positions = self._insert_molecule_at_position(
            interface_structure.positions,
            self._parsed_molecule,
            wrapped_position
        )
        
        # Update molecule_index
        new_molecule_index = self._update_molecule_index(
            interface_structure.molecule_index,
            self._parsed_molecule
        )
        
        return CustomMoleculeStructure(
            positions=new_positions,
            cell=interface_structure.cell,
            molecule_index=new_molecule_index,
            custom_molecule_count=1,
            custom_molecule_name=self._parsed_molecule['name'],
            source_structure=interface_structure
        )
    
    def _apply_pbc(self, position: np.ndarray, cell: np.ndarray) -> np.ndarray:
        """
        Wrap position into primary unit cell using PBC.
        
        cell is (3, 3) array with lattice vectors as ROW vectors.
        To wrap: convert to fractional coords, wrap to [0, 1), convert back.
        """
        # Inverse cell matrix for fractional coordinates
        inv_cell = np.linalg.inv(cell.T)  # Note: transpose for column vectors
        
        # Convert to fractional coordinates
        frac_coords = position @ inv_cell
        
        # Wrap to [0, 1)
        frac_coords = frac_coords % 1.0
        
        # Convert back to Cartesian
        wrapped_position = frac_coords @ cell.T
        
        return wrapped_position
```

## PBC Handling for Custom Molecules

### Problem

When users upload custom molecules, the coordinates may span periodic boundaries. We need to:
1. Detect if molecule is "broken" across PBC
2. Unwrap the molecule to make it whole
3. Re-wrap the molecule into the primary unit cell

### Solution

```python
# In custom_molecule_inserter.py

def _unwrap_molecule(self, positions: np.ndarray, cell: np.ndarray) -> np.ndarray:
    """
    Unwrap molecule that may be broken across periodic boundaries.
    
    Strategy:
    1. For each atom, check distance to previous atom
    2. If distance > half box length, shift by cell vector
    3. Apply recursively to make molecule whole
    """
    unwrapped = positions.copy()
    
    for i in range(1, len(positions)):
        # Vector from previous atom to current
        delta = unwrapped[i] - unwrapped[i-1]
        
        # Check if jump is too large (crossing PBC)
        # Use fractional coordinates for proper PBC handling
        inv_cell = np.linalg.inv(cell.T)
        frac_delta = delta @ inv_cell
        
        # Shift by integer multiples of cell vectors
        for j in range(3):
            if abs(frac_delta[j]) > 0.5:
                # Shift back
                shift = np.round(frac_delta[j])
                unwrapped[i:] -= shift * cell[j]
    
    return unwrapped

def _wrap_into_cell(self, positions: np.ndarray, cell: np.ndarray) -> np.ndarray:
    """
    Wrap positions into primary unit cell.
    
    Uses fractional coordinates for proper triclinic cell handling.
    """
    inv_cell = np.linalg.inv(cell.T)
    frac_coords = positions @ inv_cell
    
    # Wrap to [0, 1)
    frac_coords = frac_coords % 1.0
    
    # Convert back to Cartesian
    return frac_coords @ cell.T

def _center_molecule(self, positions: np.ndarray) -> np.ndarray:
    """Center molecule at origin (geometric center)."""
    center = positions.mean(axis=0)
    return positions - center

def _place_molecule(
    self, 
    molecule_positions: np.ndarray, 
    target_position: np.ndarray,
    cell: np.ndarray
) -> np.ndarray:
    """
    Place molecule at target position with PBC handling.
    
    Steps:
    1. Center molecule at origin
    2. Translate to target position
    3. Wrap into primary unit cell
    """
    # Center molecule
    centered = self._center_molecule(molecule_positions)
    
    # Place at target
    placed = centered + target_position
    
    # Wrap into cell
    wrapped = self._wrap_into_cell(placed, cell)
    
    return wrapped
```

```python
# In custom_molecule_inserter.py

def _parse_gro_file(self, gro_path: str) -> dict:
    """Parse GROMACS .gro file."""
    
    with open(gro_path, 'r') as f:
        lines = f.readlines()
    
    # Line 1: title
    # Line 2: number of atoms
    n_atoms = int(lines[1].strip())
    
    # Lines 3 to 3+n_atoms-1: atom records
    positions = []
    atom_names = []
    res_names = []
    res_nums = []
    
    for i in range(2, 2 + n_atoms):
        line = lines[i]
        # GRO format: resnum(5) resname(5) atomname(5) atomnum(5) x(8) y(8) z(8)
        res_num = int(line[0:5])
        res_name = line[5:10].strip()
        atom_name = line[10:15].strip()
        x = float(line[20:28]) * 0.1  # nm
        y = float(line[28:36]) * 0.1
        z = float(line[36:44]) * 0.1
        
        res_nums.append(res_num)
        res_names.append(res_name)
        atom_names.append(atom_name)
        positions.append([x, y, z])
    
    # Last lines: box vectors
    box_line = lines[-1].strip().split()
    box_vectors = [float(x) for x in box_line]
    
    return {
        'atom_count': n_atoms,
        'positions': np.array(positions),
        'atom_names': atom_names,
        'res_names': res_names,
        'res_nums': res_nums,
        'box_vectors': box_vectors
    }

def _parse_itp_file(self, itp_path: str) -> dict:
    """Parse GROMACS .itp file for [ moleculetype ] and [ atoms ] sections."""
    
    with open(itp_path, 'r') as f:
        content = f.read()
    
    # Parse [ moleculetype ]
    moleculetype_name = None
    for line in content.split('\n'):
        line = line.strip()
        if line.startswith('[') and 'moleculetype' in line:
            # Next non-comment, non-empty line has the name
            continue
        if moleculetype_name is None and line and not line.startswith(';'):
            parts = line.split()
            if len(parts) >= 1:
                moleculetype_name = parts[0]
                break
    
    # Parse [ atoms ]
    atom_types = []
    charges = []
    masses = []
    
    in_atoms_section = False
    for line in content.split('\n'):
        stripped = line.strip()
        
        if stripped.startswith('['):
            section = stripped.strip('[] \t')
            in_atoms_section = (section == 'atoms')
            continue
        
        if in_atoms_section and stripped and not stripped.startswith(';'):
            parts = stripped.split()
            if len(parts) >= 8:
                # Format: nr type resi res atom cgnr charge mass
                atom_types.append(parts[1])
                charges.append(float(parts[6]))
                masses.append(float(parts[7]))
    
    return {
        'molecule_name': moleculetype_name,
        'atom_count': len(atom_types),
        'atom_types': atom_types,
        'charges': charges,
        'masses': masses,
        'residue_name': self._get_residue_name_from_itp(content)
    }

def _get_residue_name_from_itp(self, content: str) -> str:
    """Extract residue name from [ atoms ] section."""
    for line in content.split('\n'):
        stripped = line.strip()
        if stripped and not stripped.startswith(';') and not stripped.startswith('['):
            parts = stripped.split()
            if len(parts) >= 4:
                return parts[3]  # resname field
    
    return "UNK"

def _validate_consistency(self, gro_data: dict, itp_data: dict) -> tuple[bool, str]:
    """
    Validate consistency between .gro and .itp files.
    
    Returns:
        (valid, error_message)
    """
    # Check atom count
    if gro_data['atom_count'] != itp_data['atom_count']:
        return False, (
            f"Atom count mismatch: .gro has {gro_data['atom_count']} atoms, "
            f".itp has {itp_data['atom_count']} atoms"
        )
    
    # Check for valid moleculetype name (no spaces, reasonable length)
    mol_name = itp_data['molecule_name']
    if not mol_name or ' ' in mol_name or len(mol_name) > 10:
        return False, (
            f"Invalid moleculetype name: '{mol_name}'. "
            "Must be non-empty, no spaces, max 10 characters."
        )
    
    # Check that charges are reasonable
    total_charge = sum(itp_data['charges'])
    if abs(total_charge) > 0.1:  # Allow small floating point errors
        # Warn but don't fail (charged molecules are valid)
        pass  # Could log warning
    
    # Check that all atom types are defined
    # (would need to check against force field, defer to grompp)
    
    return True, ""
```

## Anti-Patterns to Avoid

### Anti-Pattern 1: Per-Tab ViewModels

**What:** Creating separate ViewModel classes for each tab

**Why bad:** 
- Duplicates worker management code
- Fragments state management
- Complicates MainWindow coordination

**Instead:** Use single MainViewModel with signals for each tab

```python
# BAD
class SoluteViewModel(QObject):
    def start_generation(self, ...):
        # Duplicate worker management code

# GOOD
class MainViewModel(QObject):
    # Centralized worker management
    def start_solute_generation(self, ...):
    def start_custom_molecule_generation(self, ...):
```

### Anti-Pattern 2: Inline File Parsing in UI

**What:** Parsing .gro/.itp files directly in Panel classes

**Why bad:**
- UI code becomes complex and hard to test
- No reuse for other components
- Validation logic mixed with UI logic

**Instead:** Create dedicated Inserter class with parse/validate methods

```python
# BAD
class CustomMoleculePanel(QWidget):
    def _on_upload_gro(self):
        # 50 lines of parsing code...

class CustomMoleculePanel(QWidget):
    def _on_upload_gro(self):
        # Just trigger validation worker

        self._validation_worker.start()

class CustomMoleculeInserter:
    def validate_and_parse(self):
        # All parsing/validation logic here
```

### Anti-Pattern 3: Shared Mutable Registry

**What:** Using singleton MoleculetypeRegistry that persists across generations

**Why bad:**
- Stale entries from previous generations
- Name collisions between different sessions
- Hard to reset state

**Instead:** Create fresh registry per generation or use context manager

```python

        return cls._instance

        self._custom_counter = 0
        self._initialize_bundled()

```

### Anti-Pattern 4: Ignoring PBC in Custom Molecules

**What:** Assuming uploaded .gro coordinates are already in primary cell

**Why bad:**
- Custom molecules may span periodic boundaries
- Placement may put molecule outside cell
- GROMACS requires all atoms in primary cell

**Instead:** Always apply PBC wrapping after placement

```python
# BAD
def insert_molecule(self, position):
    return molecule_positions + position

# GOOD
def insert_molecule(self, position, cell):
    placed = molecule_positions + position
    return self._wrap_into_cell(placed, cell)
```

### Anti-Pattern 5: Hardcoded Moleculetype Names

**What:** Using fixed names like "CH4" for all methane molecules

**Why bad:**
- GROMACS requires unique moleculetype names
- CH4 in hydrate vs CH4 in liquid have different contexts
- Conflicts in topology file

**Instead:** Use context-aware naming via MoleculetypeRegistry

```python
# BAD
# In hydrate generator
moleculetype = "CH4"

# In solute inserter
moleculetype = "CH4"  # CONFLICT!

# GOOD
# In hydrate generator
registry.register_hydrate_guest("ch4")  # Returns "CH4_HYD"

# In solute inserter

```
